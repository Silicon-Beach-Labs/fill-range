## Usage

```js
var range = require('{%= name %}');
```

**Params**

```js
range(start, stop, increment);
```

 - `start`: the number or letter to start with
 - `end`: the number or letter to end with
 - `increment`: optionally pass the increment to use. works for letters or numbers

**Examples**

```js
range(1, 3)
//=> ['1', '2', '3']

range('1', '3')
//=> ['1', '2', '3']

range('0', '-5')
//=> [ '0', '-1', '-2', '-3', '-4', '-5' ]

range(-9, 9, 3)
//=> [ '-9', '-6', '-3', '0', '3', '6', '9' ])

range('-1', '-10', '-2')
//=> [ '-1', '-3', '-5', '-7', '-9' ]

range('1', '10', '2')
//=> [ '1', '3', '5', '7', '9' ]

range('a', 'e')
//=> ['a', 'b', 'c', 'd', 'e']

range('a', 'e', 2)
//=> ['a', 'c', 'e']

range('A', 'E', 2)
//=> ['A', 'C', 'E']
```

### Custom function

Optionally pass a custom function as the third or fourth argument:

```js
range('a', 'e', function (val, isLetter, i) {
  if (isLetter) {
    return String.fromCharCode(val) + i;
  }
  return val;
});
//=> ['a0', 'b1', 'c2', 'd3', 'e4']
```
